public class WorkerManager33 {
    public static void menu(Integer choice, String name, String email, Date birthday, Integer workerid, String department){
        if(choice==1){
            //insert
            Worker3__c workerselect= new Worker3__c();
            workerselect.Name= name;
            workerselect.Email__c= email;
            workerselect.Birthdate__c= birthday;
            workerselect.Workerid__c= workerid;
            workerselect.Department__c= department;
            insert workerselect;
            System.debug('Inserted record');
        }
        else if(choice==2){
            List<Worker3__c> workerlist;
            workerlist= [SELECT Name, Email__c, Birthdate__c, Workerid__c, Department__c FROM Worker3__c];
            for(Worker3__c workie: workerlist){
                System.debug('Name '+'|'+ workie.Name + 'Employeeid: ' + '|' + workie.Workerid__c );
            }
            //view
        }
        else if(choice==3){
            //update
            Worker3__c worker_u= [SELECT Id, Name, Email__c, Birthdate__c, Workerid__c, Department__c FROM Worker3__c WHERE Workerid__c=: workerid];
            if(name!=null){
                worker_u.Name= name;
            }
            if(department!=null){
                worker_u.Department__c= department;
            }
            update worker_u;
            System.debug('updated record');
        }
        else if(choice==4){
            //delete
            Worker3__c worker2= [SELECT Id FROM Worker3__c WHERE Workerid__c= :workerid];
            delete worker2;
        }
        else{
            System.debug('Invalid choice.');
        }
    }
}
